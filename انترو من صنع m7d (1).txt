local LoadingConfig = {
    Duration = 8,
    FadeOutTime = 1.2,

    ParticleSpeed = 0.003,

local AnimationConnections = {}
local Particles = {}

local function CreateParticleSystem(parent)
    local particleFrame = Instance.new("Frame", parent)
    particleFrame.Name = "ParticleFrame"
    particleFrame.BackgroundTransparency = 1
    particleFrame.Size = UDim2.new(1, 0, 1, 0)
    particleFrame.ClipsDescendants = true

    for i = 1, LoadingConfig.ParticleCount do
        local particle = Instance.new("Frame")
        particle.Parent = particleFrame
        particle.BackgroundColor3 = LoadingConfig.BarColor
        particle.BorderSizePixel = 0
        local size = math.random(2, LoadingConfig.ParticleSizeMax)
        particle.Size = UDim2.new(0, size, 0, size)
        particle.Position = UDim2.new(math.random(), 0, math.random(), 0)
        particle.Transparency = math.random(4, 8) / 10
        local corner = Instance.new("UICorner", particle)
        corner.CornerRadius = UDim.new(1, 0)
        
        local particleData = {
            instance = particle,
            speed = LoadingConfig.ParticleSpeed * (size / LoadingConfig.ParticleSizeMax),
            mode = "fall"
        }
        table.insert(Particles, particleData)

        local connection = RunService.Heartbeat:Connect(function()
            if particle.Parent then
                if particleData.mode == "fall" then
                    particle.Position = particle.Position + UDim2.new(0, 0, particleData.speed, 0)
                    if particle.Position.Y.Scale > 1.1 then
                        particle.Position = UDim2.new(math.random(), 0, -0.1, 0)
                    end
                elseif particleData.mode == "rise" then
                    particle.Position = particle.Position - UDim2.new(0, 0, particleData.speed, 0)
                    if particle.Position.Y.Scale < -0.1 then
                        particle.Position = UDim2.new(math.random(), 0, 1.1, 0)
                    end
                elseif particleData.mode == "blackhole" then
                    local center = Vector2.new(0.5, 0.5)
                    local currentPos = Vector2.new(particle.Position.X.Scale, particle.Position.Y.Scale)
                    local direction = (center - currentPos).Unit
                    local speedFactor = 0.005
                    particle.Position = particle.Position + UDim2.new(direction.X * speedFactor, 0, direction.Y * speedFactor, 0)
                    
                    if (center - currentPos).Magnitude < 0.05 then
                        particle.Position = UDim2.new(math.random(0, 100) / 100, 0, math.random(0,1) == 0 and -0.1 or 1.1, 0)
                    end
                end
            end
        end)
        table.insert(AnimationConnections, connection)
    end
end

local function StartLoadingSequence()
    local screenGui, progressBar, percentageLabel, statusLabel, mainFrame, shineEffect, title, loadingFrame, liveInfoLabel, versionLabel, easterEggButton, antiGravityButton, blackHoleButton = CreateLoadingScreen()
    
    local powerUpSound = Instance.new("Sound", screenGui); powerUpSound.SoundId = "rbxassetid://136923477"; powerUpSound.Volume = 0.4;
    local successSound = Instance.new("Sound", screenGui); successSound.SoundId = "rbxassetid://102221239"; successSound.Volume = 0.6;
    local whooshSound = Instance.new("Sound", screenGui); whooshSound.SoundId = "rbxassetid://162334996"; whooshSound.Volume = 0.7;
    local clickSound = Instance.new("Sound", screenGui); clickSound.SoundId = "rbxassetid://6333835654"; clickSound.Volume = 0.5;
    local gravitySwitchSound = Instance.new("Sound", screenGui); gravitySwitchSound.SoundId = "rbxassetid://2488314155"; gravitySwitchSound.Volume = 0.6;
    local blackHoleSound = Instance.new("Sound", screenGui); blackHoleSound.SoundId = "rbxassetid://1843520836"; blackHoleSound.Volume = 0.5;

    local currentParticleMode = "fall"
    antiGravityButton.MouseButton1Click:Connect(function()
        gravitySwitchSound:Play()
        if currentParticleMode == "fall" then
            currentParticleMode = "rise"
            antiGravityButton.Text = "↓"
        else
            currentParticleMode = "fall"
            antiGravityButton.Text = "↑"
        end
        for _, p_data in ipairs(Particles) do
            p_data.mode = currentParticleMode
        end
    end)

    blackHoleButton.MouseButton1Click:Connect(function()
        blackHoleSound:Play()
        if currentParticleMode == "blackhole" then
            currentParticleMode = "fall"
            antiGravityButton.Text = "↑"
        else
            currentParticleMode = "blackhole"
        end
        for _, p_data in ipairs(Particles) do
            p_data.mode = currentParticleMode
        end
    end)

    local rgbMode = 0
    local rgbConnection = nil

    local function setAllColors(color)
        mainFrame.UIStroke.Color = color
        title.UIGradient.Color = ColorSequence.new{ColorSequenceKeypoint.new(0, color), ColorSequenceKeypoint.new(1, Color3.new(1,1,1))}
        progressBar.BackgroundColor3 = color
        progressBar.UIGradient.Color = ColorSequence.new{ColorSequenceKeypoint.new(0, Color3.fromHSV(color:ToHSV() + 0.1, 1, 1)), ColorSequenceKeypoint.new(1, color)}
        for _, p in ipairs(Particles) do
            p.instance.BackgroundColor3 = color
        end
        easterEggButton.BackgroundColor3 = color
        antiGravityButton.BackgroundColor3 = color
        blackHoleButton.BackgroundColor3 = color
    end

    local function resetColors()
        mainFrame.UIStroke.Color = Color3.fromRGB(138, 43, 226)
        title.UIGradient.Color = ColorSequence.new{ColorSequenceKeypoint.new(0.00, Color3.fromRGB(220, 180, 255)), ColorSequenceKeypoint.new(1.00, Color3.fromRGB(150, 70, 255))}
        progressBar.BackgroundColor3 = LoadingConfig.BarColor
        progressBar.UIGradient.Color = ColorSequence.new{ColorSequenceKeypoint.new(0.00, Color3.fromRGB(180, 100, 255)), ColorSequenceKeypoint.new(1.00, Color3.fromRGB(110, 30, 200))}
        for _, p in ipairs(Particles) do
            p.instance.BackgroundColor3 = LoadingConfig.BarColor
        end
        easterEggButton.BackgroundColor3 = Color3.fromRGB(138, 43, 226)
        antiGravityButton.BackgroundColor3 = Color3.fromRGB(138, 43, 226)
        blackHoleButton.BackgroundColor3 = Color3.fromRGB(138, 43, 226)
    end

    easterEggButton.MouseButton1Click:Connect(function()
        clickSound:Play()
        rgbMode = (rgbMode + 1) % 5
        
        if rgbConnection then rgbConnection:Disconnect(); rgbConnection = nil end

        if rgbMode == 0 then
            resetColors()
        elseif rgbMode == 1 then
            rgbConnection = RunService.Heartbeat:Connect(function() setAllColors(Color3.fromHSV(tick() % 5 / 5, 1, 1)) end)
        elseif rgbMode == 2 then
            rgbConnection = RunService.Heartbeat:Connect(function() setAllColors(Color3.fromRGB(138, 43, 226):Lerp(Color3.fromRGB(255, 105, 180), (math.sin(tick() * 4) + 1) / 2)) end)
        elseif rgbMode == 3 then
            rgbConnection = RunService.Heartbeat:Connect(function()
                local colors = {Color3.fromRGB(138, 43, 226), Color3.fromRGB(0, 191, 255), Color3.fromRGB(75, 0, 130)}
                local segment = math.floor((tick() * 0.5) % #colors) + 1
                setAllColors(colors[segment]:Lerp(colors[(segment % #colors) + 1], (tick() * 0.5) % 1))
            end)
        elseif rgbMode == 4 then
            local baseColor = Color3.fromRGB(138, 43, 226)
            local brightColor = Color3.fromRGB(220, 180, 255)
            rgbConnection = RunService.Heartbeat:Connect(function() setAllColors(baseColor:Lerp(brightColor, (math.sin(tick() * 2) + 1) / 2)) end)
        end
        if rgbConnection then table.insert(AnimationConnections, rgbConnection) end
    end)

    TweenService:Create(screenGui.Background, TweenInfo.new(0.5), {BackgroundTransparency = 0}):Play()
    powerUpSound:Play()

    local entranceTween = TweenService:Create(mainFrame, TweenInfo.new(0.8, Enum.EasingStyle.Elastic, Enum.EasingDirection.Out), {Size = UDim2.new(0, 480, 0, 220)})
    entranceTween:Play()
    entranceTween.Completed:Connect(function()
        mainFrame.ClipsDescendants = false
        TweenService:Create(title, TweenInfo.new(0.4), {TextTransparency = 0}):Play()
        TweenService:Create(liveInfoLabel, TweenInfo.new(0.4), {TextTransparency = 0}):Play()
        TweenService:Create(loadingFrame, TweenInfo.new(0.4), {BackgroundTransparency = 0}):Play()
        TweenService:Create(statusLabel, TweenInfo.new(0.4), {TextTransparency = 0}):Play()
        TweenService:Create(percentageLabel, TweenInfo.new(0.4), {TextTransparency = 0}):Play()
        TweenService:Create(versionLabel, TweenInfo.new(0.4), {TextTransparency = 0}):Play()
        TweenService:Create(easterEggButton, TweenInfo.new(0.4), {BackgroundTransparency = 0.8}):Play()
        TweenService:Create(antiGravityButton, TweenInfo.new(0.4), {BackgroundTransparency = 0.8}):Play()
        TweenService:Create(blackHoleButton, TweenInfo.new(0.4), {BackgroundTransparency = 0.8}):Play()
    end)

    local pulseTween = TweenService:Create(mainFrame, TweenInfo.new(2.5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {Size = UDim2.new(0, 485, 0, 225)})
    local shineTween = TweenService:Create(shineEffect, TweenInfo.new(1.5, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut, -1), {Position = UDim2.new(1, 0, 0, 0)})
    
    wait(1)
    pulseTween:Play()
    shineTween:Play()
    table.insert(AnimationConnections, pulseTween)
    table.insert(AnimationConnections, shineTween)

    local startTime = tick()
    local lastInfoUpdateTime = 0
    
    local updateConnection
    updateConnection = RunService.Heartbeat:Connect(function()
        local elapsed = tick() - startTime
        local progress = math.min(elapsed / LoadingConfig.Duration, 1)
        local percentage = math.floor(progress * 100)
        
        progressBar.Size = UDim2.new(progress, 0, 1, 0)
        percentageLabel.Text = percentage .. "%"
        
        if tick() - lastInfoUpdateTime > 2 then
            local playerCount = #Players:GetPlayers()
            local ping = math.floor(game:GetService("Stats").Network.ServerStatsItem["Data Ping"]:GetValue())
            local ownerName = "N/A"
            pcall(function() ownerName = Players:GetNameFromUserIdAsync(game.CreatorId) end)
            
            local infoToShow = {
                "عدد اللاعبين: " .. playerCount,
                "البنق: " .. ping .. "ms",
                "مالك السيرفر: " .. ownerName
            }
            liveInfoLabel.Text = infoToShow[math.random(1, #infoToShow)]
            lastInfoUpdateTime = tick()
        end
        
        if progress >= 1 then
            updateConnection:Disconnect()
            
            successSound:Play()
            wait(0.8)
            
            for _, connection in pairs(AnimationConnections) do
                if typeof(connection) == "RBXScriptConnection" then
                    connection:Disconnect()
                end
                if typeof(connection) == "Tween" then
                    connection:Cancel()
                end
            end
            
            whooshSound:Play()
            
            mainFrame.ClipsDescendants = true
            TweenService:Create(title, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            TweenService:Create(liveInfoLabel, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            TweenService:Create(loadingFrame, TweenInfo.new(0.3), {BackgroundTransparency = 1}):Play()
            TweenService:Create(statusLabel, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            TweenService:Create(percentageLabel, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            TweenService:Create(versionLabel, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            TweenService:Create(easterEggButton, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            TweenService:Create(antiGravityButton, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            TweenService:Create(blackHoleButton, TweenInfo.new(0.3), {TextTransparency = 1}):Play()
            
            wait(0.3)

            local exitTween = TweenService:Create(mainFrame, TweenInfo.new(0.8, Enum.EasingStyle.Elastic, Enum.EasingDirection.In), {Size = UDim2.new(0, 0, 0, 0)})
            local backgroundFade = TweenService:Create(screenGui.Background, TweenInfo.new(LoadingConfig.FadeOutTime), {BackgroundTransparency = 1})
            
            exitTween:Play()
            backgroundFade:Play()
            
            exitTween.Completed:Connect(function()
                screenGui:Destroy()
            end)
        end
    end)
end

StartLoadingSequence()

wait(LoadingConfig.Duration + LoadingConfig.FadeOutTime)
